! Fibonacci Sequence for Anato-CPU
! Setup initial values
MOV 0, R3      ! a = 0
MOV 1, R4      ! b = 1

ADD R3, R4, R6 ! R6 = a + b
MOV R4, R3     ! a = b
MOV R6, R4     ! b = R6

MOV R3, @100   ! Copy 'a' to RAM address 100 so we can print it
MOV 100, R0    ! Syscall Arg: Address 100
MOV 1, R1      ! Syscall Arg: Length 1
MOV 2, R2      ! Syscall Arg: Mode 2 (Integer)
INT 0          ! Call Print

HLT
